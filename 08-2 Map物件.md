# 08-2 Map物件

```
Map 物件存放的資料以(key, value)型式存在.
```


### 程式範例

```javascript
//------------------------------------------
// 使用嚴格模式
//------------------------------------------
'use strict';

//------------------------------------------
// 建立一個Map物件 (以key, value方式存放資料)
//------------------------------------------
const map = new Map();

//------------------------------------------
// 加入資料
//------------------------------------------
map.set('台北', 1);
map.set('台北', 2);
map.set('台中', 1);
map.set('台中', 2);
map.set('台中', 3);
map.set('高雄', 1);
map.set('高雄', 2);


var v = map.get('台南');

if(v == undefined){
var v = map.get('台南');
    map.set('台南', 1);
}else{
    map.set('台南', v+1);
}

console.log(map.get('台南'));

//------------------------------------------
// 印出資料
//------------------------------------------
console.log('資料個數:' + map.size);

map.forEach((k, v) => { 
    console.log(k + '->' + v);
});

//------------------------------------------
// 刪除資料
//------------------------------------------
map.delete('台北');
map.delete('台中');

//------------------------------------------
// 印出資料
//------------------------------------------
console.log('資料個數:' + map.size);

map.forEach((k, v) => { 
    console.log(k + '->' + v);
});

//------------------------------------------
// 印出資料
//------------------------------------------
console.log('map中是否有10?' + map.has('台北'));
console.log('map中是否有20?' + map.has('台中'));
console.log('map中是否有30?' + map.has('台南'));
console.log('map中是否有30?' + map.has('高雄'));
```
