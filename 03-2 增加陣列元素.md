# 03-2 增加陣列元素

### 將元素加入陣列的前端, 尾端及其中指定的位置

```javascript
//-----------------------------------
// 使用嚴格模式
//-----------------------------------
'use strict';

//-----------------------------------
// 印出陣列內容
//-----------------------------------
function printArray(array){
    for(let data of array){
        console.log(data);
    }   

    console.log();
}

//--------------------------------------
// 宣告一個Array
//--------------------------------------
var students = ['蔡家雅', '張怡宇', '王庭棋', '黃和夫', '林于文'];

//-----------------------------------
// 增加Array內容 (加入尾端)
//-----------------------------------
students.push('唐馨馨');
printArray(students);

//-----------------------------------
// 增加Array內容 (加入開頭端)
//-----------------------------------
students.unshift('應立煌');
printArray(students);

//-----------------------------------
// 增加Array內容 (加入index=2的位置)
//-----------------------------------
students.splice(2, 0, "張韻庭");
printArray(students);
```
