#第08課 關於建構元函式(2)

## (1) 建構元函式內不可直接取用的區域變數


#####測試程式
```javascript
//===========================
// 用建構元函式生成物件s1
//===========================
var s1=new Student('10456001', '王小明');   //註1

console.log(s1.getStuNo());
console.log(s1.getStuName());
console.log(s1.dept());
console.log('--------------');

//-------------------------------
// s1內的區域變數不可直接取存
//-------------------------------
console.log(s1.stuNo);   
console.log(s1.stuName);
console.log('--------------');


//===========================
// 用建構元函式生成物件s2
//===========================
var s2=new Student('10456002', '陳小華');
console.log(s2.getStuNo());
console.log(s2.getStuName());
console.log(s2.dept());
console.log('--------------');



//================================================
// 一建構元函式，用來生成物件, 建議以大寫字開頭
//================================================
function Student(stuNo, stuName){
    //------------------------
    // 判斷輸入參數個數及型態
    //------------------------
    if(arguments.length!=2){
        return null;
    }

    if(typeof(arguments[0])!='string'){
        return null;
    }
    
    if(arguments[0].length!=8){
        return null;
    }  
    
    //------------------------
    // 建立物件內區域變數
    //------------------------    
    var stuNo=stuNo;
    var stuName=stuName;

    //------------------------
    // 建立物件內函式屬性
    //------------------------
    this.getStuNo=function(){return stuNo;}
    this.getStuName=function(){return stuName;}
    
    this.dept=function(){
        var d='未知';
        
        var s=stuNo.charAt(3);
        if(s=='3'){
            d='二技';
        }else if(s=='4'){
            d='四技';
        }else if(s=='5'){
            d='五專';
        }
        
        return d;
    }
}

/*註1 的執行步驟:
(1)建立一個s1物件.
(2)設定this=s1, 再執行Student()函式.
(3)在Student()函式中, var stuNo建立了s1的stuNo區域變數, 再賦予傳入變數stuNo的值.
*/
```



#####執行結果
```
10456001
王小明
五專
--------------
undefined
undefined
--------------
10456002
陳小華
五專
--------------
```



## (2) 建構元函式內不可直接取用的區域變數


#####測試程式
```javascript
//===========================
// 用建構元函式生成物件s1
//===========================
var s1=new Student('10456001', '王小明');   //註1

console.log(s1.getStuNo());
console.log(s1.getStuName());
console.log(s1.dept());
console.log('--------------');

//-------------------------------
// s1內的區域變數不可直接取存
//-------------------------------
console.log(s1.stuNo);   
console.log(s1.stuName);
console.log(s1.deptNo);
console.log('--------------');


//===========================
// 用建構元函式生成物件s2
//===========================
var s2=new Student('10456002', '陳小華');
console.log(s2.getStuNo());
console.log(s2.getStuName());
console.log(s2.dept());
console.log('--------------');



//================================================
// 一建構元函式，用來生成物件, 建議以大寫字開頭
//================================================
function Student(stuNo, stuName){
    //------------------------
    // 判斷輸入參數個數及型態
    //------------------------
    if(arguments.length!=2){
        return null;
    }

    if(typeof(arguments[0])!='string'){
        return null;
    }
    
    if(arguments[0].length!=8){
        return null;
    }  
    
    //------------------------
    // 建立物件內區域變數
    //------------------------    
    var stuNo=stuNo;
    var stuName=stuName;
    
    var deptNo=function(){
        return stuNo.charAt(3);
    }
    
    //------------------------
    // 建立物件內函式屬性
    //------------------------
    this.getStuNo=function(){return stuNo;}
    this.getStuName=function(){return stuName;}
    
    this.dept=function(){
        var d='未知';
        
        var s=deptNo();
        if(s=='3'){
            d='二技';
        }else if(s=='4'){
            d='四技';
        }else if(s=='5'){
            d='五專';
        }
        
        return d;
    }
}
```



#####執行結果
