# 10 函式作為參數及回傳值

### (1) 函式作為參數

```javascript
//--------------------------------------
// 使用嚴格模式, 使用的變數一定要事前宣告
//--------------------------------------
'use strict';

//----------------------------------------- 
// 第1個參數是一個函式
//----------------------------------------- 
function sum(f, ...values){
    var tot = 0;

    for(let v of values){
        tot += f(v);
    }

    return tot;
}

//------------------------------
// 主程式
//------------------------------
console.log(sum(x=>x*x, 2, 3, 4, 5));    //54
```


### (2) 函式作為回傳值

```javascript
//--------------------------------------
// 使用嚴格模式, 使用的變數一定要事前宣告
//--------------------------------------
'use strict';

//----------------------------------------- 
// 第1個參數是一個函式
//----------------------------------------- 
function func(i){
    if(i==1){
        return x => x*x;
    }else{
        return x => x*x*x;
    }
}

//------------------------------
// 主程式
//------------------------------
var f = func(1);    
console.log(f(10));   //100

var f = func(2);    
console.log(f(10));   //1000
```
