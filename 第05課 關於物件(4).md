## (1) 物件的getter及setter

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={
    cnt:0,

    // 建立一個getter
    get getCounter(){
    	this.cnt++;
    	return this.cnt;
    },
    
	// 建立一個setter
    set setCounter(c){
    	this.cnt=c;
    },
    
    a:function(){return 'Hello';}
}


//===============================
// 呼叫物件m的getCounter()方法
//===============================
console.log(m.getCounter);
console.log(m.getCounter);

//==================================
// 呼叫物件m的setCounter(...)方法
//==================================
m.setCounter=10;

//===============================
// 呼叫物件m的getCounter()方法
//===============================
console.log(m.getCounter);
console.log(m.getCounter);
```



#####執行結果
```
1
2
11
12
```



## (2) 呼叫物件內函式的call(...)方法

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={
    stuNo:'1456001',
    name:'王小明',
    gender:'M',

    getStuNo:function(){return this.stuNo;},
    getName:function(){return this.name;},
    
    //-----------------
    // 回傳中文性別
    //-----------------
    getGender:function(){
        var gender='有誤';
        
        if(this.gender=='M'){
            gender='男';
        }else{
            gender='女';
        }
        
        return gender;
    }
}


//====================
// 呼叫物件中的方法
//====================
console.log(m.getStuNo());
console.log(m.getName());

//-----------------------------------------
// 使用call(...)方法, 呼叫某物件中的函式
//-----------------------------------------
var parameter={gender:'F'};
console.log(m.getGender.call(parameter));
```



#####執行結果
```
1456001
王小明
女
```
