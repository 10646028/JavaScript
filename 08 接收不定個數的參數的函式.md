# 08 接收不定個數的參數的函式

### (1) 範例

```javascript
//--------------------------------------
// 使用嚴格模式, 使用的變數一定要事前宣告
//--------------------------------------
'use strict';

//----------------------------------------- 
// 可接收不定個數參數的函式
//----------------------------------------- 
function add(...values) {
    var total = 0;
  
    for(let val of values) {
        total += val;
    }
  
    return total;
}

//------------------------------
// 主程式
//------------------------------
console.log(add(10, 15, 13));       //38
console.log(add(10, 15, 13, 24));   //62
```


### (2) 範例

```javascript
//--------------------------------------
// 使用嚴格模式, 使用的變數一定要事前宣告
//--------------------------------------
'use strict';

//----------------------------------------- 
// 可接收不定個數參數的函式
//----------------------------------------- 
function add(minVal, ...values) {
    var total = 0;
  
    for(let val of values) {
        if(val >= minVal){
            total += val;
        }    
    }
  
    return total;
}

//------------------------------
// 主程式
//------------------------------
console.log(add(12, 10, 15, 13));       //28
console.log(add(20, 10, 15, 13, 24));   //24
```
