#第02課 關於物件



## (1) 建立物件

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={stuName:'王小明', age:20};


//====================
// 印出物件的型態
//====================
console.log(typeof(m));
console.log("----------------------");


//====================
// 印出物件的屬性(1)
//====================
console.log(m.stuName);
console.log(m.age);
console.log("----------------------");


//====================
// 印出物件的屬性(2)
//====================
console.log(m["stuName"]);
console.log(m["age"]);
console.log("----------------------");


//====================
// 印出物件的屬性(3)
//====================
var k1="stuName";
var k2="age";
console.log(m[k1]);
console.log(m[k2]);
console.log("----------------------");
```



#####執行結果
```
object
----------------------
王小明
20
----------------------
王小明
20
----------------------
王小明
20
----------------------
```



## (2) 刪除物件的屬性

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={stuName:'王小明', age:20};


//====================
// 印出物件的屬性
//====================
console.log(m.stuName);
console.log(m.age);
console.log("----------------------");


//====================
// 刪除物件的屬性
//====================
delete(m.stuName);
console.log(m.stuName);
console.log(m.age);
console.log("----------------------");


//=============================
// 判斷物件是否擁有某個屬性
//=============================
console.log(m.hasOwnProperty("stuName"));
console.log(m.hasOwnProperty("age"));
console.log("----------------------");
```



#####執行結果
```
王小明
20
----------------------
undefined
20
----------------------
false
true
----------------------
```



## (3) 列出物件所有的屬性

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={stuName:'王小明', age:20, gender:'男'};


//======================
// 印出物件所有的屬性
//======================
for(property in m){
    console.log('屬性=' + property + ', 值=' + m[property]);
}
```



#####執行結果
```
屬性=stuName, 值=王小明
屬性=age, 值=20
屬性=gender, 值=男
```



## (4) 物件的getter及setter

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={
    cnt:0,

    // 建立一個getter
    get getCounter(){
    	this.cnt++;
    	return this.cnt;
    },
    
	// 建立一個setter
    set setCounter(c){
    	this.cnt=c;
    },
    
    a:function(){return 'Hello';}
}


//===============================
// 呼叫物件m的getCounter()方法
//===============================
console.log(m.getCounter);
console.log(m.getCounter);

//==================================
// 呼叫物件m的setCounter(...)方法
//==================================
m.setCounter=10;

//===============================
// 呼叫物件m的getCounter()方法
//===============================
console.log(m.getCounter);
console.log(m.getCounter);
```



#####執行結果
```
1
2
11
12
```
