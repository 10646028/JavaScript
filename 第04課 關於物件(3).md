#第04課 關於物件(3)

## (1) 函式屬性

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={
    stuNo:'1456001',
    name:'王小明',
    gender:'M',

    getStuNo:function(){return this.stuNo;},
    getName:function(){return this.name;},
    
    //-----------------
    // 回傳中文性別
    //-----------------
    getGender:function(){
        var gender='有誤';
        
        if(this.gender=='M'){
            gender='男';
        }else{
            gender='女';
        }
        
        return gender;
    }
}


//====================
// 呼叫物件中的方法
//====================
console.log(m.getStuNo());
console.log(m.getName());
console.log(m.getGender());
```



#####執行結果
```
1456001
王小明
男
```



## (2) 由函式增加屬性

#####測試程式
```javascript
//====================
// 宣告一個物件
//====================
var m={
    setStuNo:function(stuNo){this.stuNo=stuNo;},
    setName:function(name){this.name=name;},
    setGender:function(gender){this.gender=gender;},

    getStuNo:function(){return this.stuNo;},
    getName:function(){return this.name;},

    //-----------------
    // 回傳中文性別
    //-----------------
    getGender:function(){
        var gender='有誤';

        if(this.gender=='M'){
            gender='男';
        }else{
            gender='女';
        }

        return gender;
    }
}


//====================
// 呼叫物件中的方法
//====================
m.setStuNo('1456002');
m.setName('陳小華');
m.setGender('F');

console.log(m.getStuNo());
console.log(m.getName());
console.log(m.getGender());
```



#####執行結果
```
1456002
陳小華
女
```
