## (1) 一個回傳相加值的函式

#####測試程式
```javascript
console.log(sum(10, 20));


//==================
// 宣告一個函式
//==================
function sum(a, b){
    return a+b;
}
```

```
30
```


## (2) 一個回傳學制的函式

#####測試程式
```javascript
console.log(division('10456001'));
console.log(division('10446001'));
console.log(division('10436001'));
console.log(division('10476001'));
console.log(division('104760012'));
console.log(division());


//===========================================
// 傳入學號, 依學號第4碼判斷學制, 回傳學制
//===========================================
function division(stuNo){
    //------------------------------
    // 判斷輸入參數文字長度及個數
    //------------------------------
    if(arguments.length!=1){
        return '參數錯誤';
    }else if(stuNo.length!=8){
        return '資料錯誤';
    }
    	
    //-----------
    // 判斷學制
    //-----------
    var deptName=null;
    var deptNo=stuNo.charAt(3);
    
    if(deptNo=='3'){
    	deptName = '二技';
    }else if(deptNo=='4'){
    	deptName = '四技';
    }else if(deptNo=='5'){
    	deptName = '五專';
    }else{
    	deptName = '資料錯誤';
    }
    
    //---------------
    // 回傳學制
    //---------------
    return deptName;
}
```



#####執行結果
```
五專
四技
二技
資料錯誤
資料錯誤
參數錯誤
```



## (3) 一個計算傭金函式

#####測試程式
```javascript
console.log(commission('A', 350000));
console.log(commission('B', 250000));
console.log(commission('A', '350000'));
console.log(commission(new String('A'), 200000));
console.log(commission(350000));


//===========================================
// 傳入職等及銷售金額, 計算並回傳傭金
//===========================================
function commission(rank, total){
    //------------------------
    // 判斷輸入參數個數及型態
    //------------------------
    if(arguments.length!=2){
        return "參數錯誤1";
    }
	
    if(typeof(rank)!='string'){
        return "資料錯誤2";
    }
    
    if(typeof(total)!='number'){
        return "資料錯誤3";
    }    
    	
    //-----------
    // 計算傭金
    //-----------
    var commission=0;
	
    if(rank=='A'){
    	commission=parseInt(total*0.12);
    }else{
    	commission=parseInt(total*0.08);
    }
    
    //---------------
    // 回傳傭金
    //---------------
    return commission;
}
```



#####執行結果
```
42000
20000
資料錯誤3
資料錯誤2
參數錯誤1
```
