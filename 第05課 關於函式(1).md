#第05課 關於函式(1)

## (1) 一個回傳相加值的函式

#####測試程式
```javascript
console.log(sum(10, 20));


//==================
// 宣告一個函式
//==================
function sum(a, b){
    return a+b;
}
```



#####執行結果
```
30
```


## (2) 一個回傳學制的函式

#####測試程式
```javascript
console.log(division('10456001'));
console.log(division('10446001'));
console.log(division('10436001'));
console.log(division('10476001'));
console.log(division('104760012'));
console.log(division());


//===========================================
// 傳入學號, 依學號第4碼判斷學制, 回傳學制
//===========================================
function division(stuNo){
    //------------------------------
    // 判斷輸入參數文字長度及個數
    //------------------------------
    if(arguments.length!=1){
        return '參數錯誤';
    }else if(stuNo.length!=8){
        return '資料錯誤';
    }
    	
    //-----------
    // 判斷學制
    //-----------
    var deptName=null;
    var deptNo=stuNo.charAt(3);
    
    if(deptNo=='3'){
    	deptName = '二技';
    }else if(deptNo=='4'){
    	deptName = '四技';
    }else if(deptNo=='5'){
    	deptName = '五專';
    }else{
    	deptName = '資料錯誤';
    }
    
    //---------------
    // 回傳學制
    //---------------
    return deptName;
}
```



#####執行結果
```
五專
四技
二技
資料錯誤
資料錯誤
參數錯誤
```
