#第01課 基本資料型別


## (1) number, string 及 boolean

#####測試程式
```javascript
//=============================
// 以 64 位元儲存的 number
//=============================
var m1=9000123456789000;
var m2=10.1*Math.PI;
console.log(m1);
console.log(m2);
console.log(typeof(m1));
console.log(typeof(m2));
console.log('--------------');


//=============================
//存字串的 string
//=============================
var m3="abc";
var m4='def\nghi';
console.log(m3);
console.log(m4);
console.log(typeof(m3));
console.log(typeof(m4));
console.log('--------------');


//=============================
//存布林值的 boolean
//=============================
var m5=true;
var m6=false;
console.log(m5);
console.log(m6);
console.log(typeof(m5));
console.log(typeof(m6));
console.log('--------------');
```



#####執行結果
```
9000123456789000
31.73008580125691
number

--------------
abc
def
ghi
string
string
--------------
true
false
boolean
boolean
--------------
```


## (2) undefined 及 null

#####測試程式
```javascript
//================================================
// undefined是Global物件的屬性, 代表「沒有定義」
//================================================
var m1;
console.log(m1);
console.log(typeof(m1));
console.log("--------------------");



//=======================================
// null是一個物件值, 代表「沒有物件」
//=======================================
var m2=null;
console.log(m2);
console.log(typeof(m2));
console.log("--------------------");
```



#####執行結果
```
undefined
undefined
--------------------
null
object
--------------------
```

## (4) 自動包裹(auto-boxing)及包裹器(wrapper)

#####測試程式
```javascript
//==============================
// m1是一個string基本資料型別
//==============================
var m1="abcde";
console.log(typeof(m1));
console.log(m1 instanceof String);
console.log("----------------------");


//==========================
// m2是一個String物件
//==========================
var m2=new String("abcde");
console.log(typeof(m2));
console.log(m2 instanceof String);
console.log("----------------------");


//============================================================
// m1需被自動包裝(auto-boxing)成包裹器型態(wrapper type),
// 也就是先包裝成String物件, 才有方法可呼叫.
//============================================================
console.log(m1.charAt(0));


//================================================
// m2本是String物件, 不需包裹, 自有方法可以呼叫
//================================================
console.log(m2.charAt(1));
console.log("----------------------");
```



#####執行結果
```
string
false
----------------------
object
true
----------------------
a
b
----------------------
```



## (5) 型別轉換

#####測試程式
```javascript
//=======================================
// 將string型別轉成number型別(取整數)
//=======================================
var m1=parseInt("1234.56");
console.log(m1);
console.log(typeof(m1));
console.log('--------------------');


//================================
// 將string型別轉成number型別
//================================
var m2=Number("123.456");
console.log(m2);
console.log(typeof(m2));
console.log('--------------------');


//=================================================
// 將string型別轉成number型別(NaN, Not a Number)
//=================================================
var m3=parseInt('abc');
console.log(m3)
console.log(typeof(m3));
console.log(isNaN(m3));
console.log('--------------------');


//===============================
// 將number型別轉成string型別
//===============================
var m4=m1.toString();
console.log(m4)
console.log(typeof(m4));
console.log('--------------------');
```



#####執行結果
```
1234
number
--------------------
123.456
number
--------------------
NaN
number
true
--------------------
1234
string
--------------------
```
