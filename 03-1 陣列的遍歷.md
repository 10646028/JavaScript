# 03-1 陣列的遍歷

### 幾種將陣列內容逐一取出的寫法

```javascript
//--------------------------------------
// 使用嚴格模式
//--------------------------------------
'use strict';

//--------------------------------------
// 宣告一個Array
//--------------------------------------
var students = ['蔡家雅', '張怡宇', '王庭棋', '黃和夫', '林于文'];

//------------------------------
// 逐個印出Array內容 - (1)
//------------------------------
for(var i=0; i<students.length; i++){
    console.log(students[i]);
}

console.log();

//------------------------------
// 逐個印出Array內容 - (2)
//------------------------------
for(var student of students){
    console.log(student);
}

console.log();

//------------------------------
// 逐個印出Array內容 - (3)
//------------------------------
students.forEach(function(element) {
    console.log(element);
});

console.log();

//------------------------------
// 逐個印出Array內容 - (4)
//------------------------------
students.forEach(function(element, index) {
    console.log(element + ' ' + index);
});

console.log();

//------------------------------
// 逐個印出Array內容 - (5)
//------------------------------
students.forEach(x => {console.log(x)});
```
